package com.example.ornekhesaplama;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

import java.util.Random;

public class MainActivity extends AppCompatActivity {

    Button buttonDongu1,buttonDongu2,buttonDongu3,buttonDongu4;
    TextView textSonuc;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });


        buttonDongu1 = findViewById(R.id.buttonDongu1);
        buttonDongu2 = findViewById(R.id.buttonDongu2);
        buttonDongu3 = findViewById(R.id.buttonDongu3);
        buttonDongu4 = findViewById(R.id.buttonDongu4);
        textSonuc = findViewById(R.id.textSonuc);

        buttonDongu1.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                textSonuc.setText("");
                int i=1;
                while(i<=10) {
                    textSonuc.append(i + "\n");
                    i++;
                }
            }
        });

        buttonDongu2.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                textSonuc.setText("");
                int i=10;
                while(i>=1) {
                    textSonuc.append(i + "\n");
                    i--;
                }
            }
        });

        buttonDongu3.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                textSonuc.setText("");
                int i=1;
                do {
                    textSonuc.append(i + "\n");
                    i++;
                } while(i<=10);
            }
        });

        buttonDongu4.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                int i=10;
                do {
                    textSonuc.append(i + "\n");
                    i--;
                } while(i>=1);
            }
        });

    }
}
